\section{Algorithms}

\begin{algorithm}
\caption{An algorithm for computing} \label{algorithm:algorithm_1}
\begin{algorithmic}[1] % [1] for line numbers
\Procedure{compute}{$x,y,z \gets 0$}
    \Require Input data
    \Ensure Output result
    \State $a \gets x$
    \State $b \gets y$
    \State $c \gets \emptyset$
    \ForAll{$item_a \in a$}
        \State $d \gets 0$
        \ForAll{$item_b \in b$}
            \If{$item_a$ = $item_b$}
                \State $d \gets d + 1$
                \State $c \gets c \cup \{ item_a \} $
            \Else
                \State $d \gets d + 0$
            \EndIf
        \EndFor
    \EndFor
    \State \Return $c,d$
\EndProcedure
\end{algorithmic}
\end{algorithm}
